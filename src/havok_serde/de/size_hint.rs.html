9#[cfg(any(feature = "std", feature = "alloc"))]
10pub fn cautious<Element>(hint: Option<usize>) -> usize {
11    const MAX_PREALLOC_BYTES: usize = 1024 * 1024;
12
13    if mem::size_of::<Element>() == 0 {
14        0
15    } else {
16        cmp::min(
17            hint.unwrap_or(0),
18            MAX_PREALLOC_BYTES / mem::size_of::<Element>(),
19        )
20    }
21}
