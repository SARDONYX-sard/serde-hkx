43/// The tag of a change.
44#[derive(Debug, PartialEq, Eq, Hash, Clone, Copy, Ord, PartialOrd)]
45#[cfg_attr(
46    feature = "serde",
47    derive(serde::Serialize, serde::Deserialize),
48    serde(rename_all = "snake_case")
49)]
50pub enum ChangeTag {
51    /// The change indicates equality (not a change)
52    Equal,
53    /// The change indicates deleted text.
54    Delete,
55    /// The change indicates inserted text.
56    Insert,
57}

62            f,
63            "{}",

66                ChangeTag::Delete => '-',
67                ChangeTag::Insert => '+',
68            }
69        )
70    }

73/// Represents the expanded [`DiffOp`] change.
74///
75/// This type is returned from [`DiffOp::iter_changes`] and
76/// [`TextDiff::iter_changes`](crate::text::TextDiff::iter_changes).
77///
78/// It exists so that it's more convenient to work with textual differences as
79/// the underlying [`DiffOp`] encodes a group of changes.
80///
81/// This type has additional methods that are only available for types
82/// implementing [`DiffableStr`](crate::text::DiffableStr).
83#[derive(Debug, PartialEq, Eq, Hash, Clone, Copy, Ord, PartialOrd)]
84#[cfg_attr(feature = "serde", derive(serde::Serialize))]
85pub struct Change<T> {
86    pub(crate) tag: ChangeTag,
87    pub(crate) old_index: Option<usize>,
88    pub(crate) new_index: Option<usize>,
89    pub(crate) value: T,
90}
91
92/// These methods are available for all change types.
93impl<T: Clone> Change<T> {
94    /// Returns the change tag.
95    pub fn tag(&self) -> ChangeTag {
96        self.tag
97    }
98
99    /// Returns the old index if available.
100    pub fn old_index(&self) -> Option<usize> {
101        self.old_index
102    }
103
104    /// Returns the new index if available.
105    pub fn new_index(&self) -> Option<usize> {
106        self.new_index
107    }
108
109    /// Returns the underlying changed value.
110    ///
111    /// Depending on the type of the underlying [`crate::text::DiffableStr`]
112    /// this value is more or less useful.  If you always want to have a utf-8
113    /// string it's best to use the [`Change::as_str`] and
114    /// [`Change::to_string_lossy`] methods.
115    pub fn value(&self) -> T {
116        self.value.clone()
117    }
118
119    /// Returns the underlying changed value as reference.
120    pub fn value_ref(&self) -> &T {
121        &self.value
122    }
123
124    /// Returns the underlying changed value as mutable reference.
125    pub fn value_mut(&mut self) -> &mut T {
126        &mut self.value
127    }
128}

130/// Utility enum to capture a diff operation.
131///
132/// This is used by [`Capture`](crate::algorithms::Capture).
133#[derive(Debug, PartialEq, Eq, Hash, Clone, Copy)]
134#[cfg_attr(
135    feature = "serde",
136    derive(serde::Serialize, serde::Deserialize),
137    serde(rename_all = "snake_case", tag = "op")
138)]
139pub enum DiffOp {
140    /// A segment is equal (see [`DiffHook::equal`])
141    Equal {
142        /// The starting index in the old sequence.
143        old_index: usize,
144        /// The starting index in the new sequence.
145        new_index: usize,
146        /// The length of the segment.
147        len: usize,
148    },
149    /// A segment was deleted (see [`DiffHook::delete`])
150    Delete {
151        /// The starting index in the old sequence.
152        old_index: usize,
153        /// The length of the old segment.
154        old_len: usize,
155        /// The starting index in the new sequence.
156        new_index: usize,
157    },
158    /// A segment was inserted (see [`DiffHook::insert`])
159    Insert {
160        /// The starting index in the old sequence.
161        old_index: usize,
162        /// The starting index in the new sequence.
163        new_index: usize,
164        /// The length of the new segment.
165        new_len: usize,
166    },
167    /// A segment was replaced (see [`DiffHook::replace`])
168    Replace {
169        /// The starting index in the old sequence.
170        old_index: usize,
171        /// The length of the old segment.
172        old_len: usize,
173        /// The starting index in the new sequence.
174        new_index: usize,
175        /// The length of the new segment.
176        new_len: usize,
177    },
178}

180/// The tag of a diff operation.
181#[derive(Debug, PartialEq, Eq, Hash, Clone, Copy, Ord, PartialOrd)]
182#[cfg_attr(
183    feature = "serde",
184    derive(serde::Serialize, serde::Deserialize),
185    serde(rename_all = "snake_case")
186)]
187pub enum DiffTag {
188    /// The diff op encodes an equal segment.
189    Equal,
190    /// The diff op encodes a deleted segment.
191    Delete,
192    /// The diff op encodes an inserted segment.
193    Insert,
194    /// The diff op encodes a replaced segment.
195    Replace,
196}
197
198impl DiffOp {
199    /// Returns the tag of the operation.
200    pub fn tag(self) -> DiffTag {
201        self.as_tag_tuple().0
202    }
203
204    /// Returns the old range.
205    pub fn old_range(&self) -> Range<usize> {
206        self.as_tag_tuple().1
207    }
208
209    /// Returns the new range.
210    pub fn new_range(&self) -> Range<usize> {
211        self.as_tag_tuple().2
212    }
213
214    /// Transform the op into a tuple of diff tag and ranges.
215    ///
216    /// This is useful when operating on slices.  The returned format is
217    /// `(tag, i1..i2, j1..j2)`:
218    ///
219    /// * `Replace`: `a[i1..i2]` should be replaced by `b[j1..j2]`
220    /// * `Delete`: `a[i1..i2]` should be deleted (`j1 == j2` in this case).
221    /// * `Insert`: `b[j1..j2]` should be inserted at `a[i1..i2]` (`i1 == i2` in this case).
222    /// * `Equal`: `a[i1..i2]` is equal to `b[j1..j2]`.
223    pub fn as_tag_tuple(&self) -> (DiffTag, Range<usize>, Range<usize>) {
224        match *self {
225            DiffOp::Equal {
226                old_index,
227                new_index,
228                len,
229            } => (
230                DiffTag::Equal,
231                old_index..old_index + len,
232                new_index..new_index + len,
233            ),
234            DiffOp::Delete {
235                old_index,
236                new_index,
237                old_len,
238            } => (
239                DiffTag::Delete,
240                old_index..old_index + old_len,
241                new_index..new_index,
242            ),
243            DiffOp::Insert {
244                old_index,
245                new_index,
246                new_len,
247            } => (
248                DiffTag::Insert,
249                old_index..old_index,
250                new_index..new_index + new_len,
251            ),
252            DiffOp::Replace {
253                old_index,
254                old_len,
255                new_index,
256                new_len,
257            } => (
258                DiffTag::Replace,
259                old_index..old_index + old_len,
260                new_index..new_index + new_len,
261            ),
262        }
263    }
264
265    /// Apply this operation to a diff hook.
266    pub fn apply_to_hook<D: DiffHook>(&self, d: &mut D) -> Result<(), D::Error> {
267        match *self {
268            DiffOp::Equal {
269                old_index,
270                new_index,
271                len,
272            } => d.equal(old_index, new_index, len),
273            DiffOp::Delete {
274                old_index,
275                old_len,
276                new_index,
277            } => d.delete(old_index, old_len, new_index),
278            DiffOp::Insert {
279                old_index,
280                new_index,
281                new_len,
282            } => d.insert(old_index, new_index, new_len),
283            DiffOp::Replace {
284                old_index,
285                old_len,
286                new_index,
287                new_len,
288            } => d.replace(old_index, old_len, new_index, new_len),
289        }
290    }

292    /// Iterates over all changes encoded in the diff op against old and new
293    /// sequences.
294    ///
295    /// `old` and `new` are two indexable objects like the types you pass to
296    /// the diffing algorithm functions.
297    ///
298    /// ```rust
299    /// use similar::{ChangeTag, Algorithm};
300    /// use similar::capture_diff_slices;
301    /// let old = vec!["foo", "bar", "baz"];
302    /// let new = vec!["foo", "bar", "blah"];
303    /// let ops = capture_diff_slices(Algorithm::Myers, &old, &new);
304    /// let changes: Vec<_> = ops
305    ///     .iter()
306    ///     .flat_map(|x| x.iter_changes(&old, &new))
307    ///     .map(|x| (x.tag(), x.value()))
308    ///     .collect();
309    /// assert_eq!(changes, vec![
310    ///     (ChangeTag::Equal, "foo"),
311    ///     (ChangeTag::Equal, "bar"),
312    ///     (ChangeTag::Delete, "baz"),
313    ///     (ChangeTag::Insert, "blah"),
314    /// ]);
315    /// ```
316    pub fn iter_changes<'lookup, Old, New, T>(
317        &self,
318        old: &'lookup Old,
319        new: &'lookup New,
320    ) -> ChangesIter<'lookup, Old, New, T>
321    where
322        Old: Index<usize, Output = T> + ?Sized,
323        New: Index<usize, Output = T> + ?Sized,
324    {
325        ChangesIter::new(old, new, *self)
326    }

328    /// Given a diffop yields the changes it encodes against the given slices.
329    ///
330    /// This is similar to [`DiffOp::iter_changes`] but instead of yielding the
331    /// individual changes it yields consequitive changed slices.
332    ///
333    /// This will only ever yield a single tuple or two tuples in case a
334    /// [`DiffOp::Replace`] operation is passed.
335    ///
336    /// ```rust
337    /// use similar::{ChangeTag, Algorithm};
338    /// use similar::capture_diff_slices;
339    /// let old = vec!["foo", "bar", "baz"];
340    /// let new = vec!["foo", "bar", "blah"];
341    /// let ops = capture_diff_slices(Algorithm::Myers, &old, &new);
342    /// let changes: Vec<_> = ops.iter().flat_map(|x| x.iter_slices(&old, &new)).collect();
343    /// assert_eq!(changes, vec![
344    ///     (ChangeTag::Equal, &["foo", "bar"][..]),
345    ///     (ChangeTag::Delete, &["baz"][..]),
346    ///     (ChangeTag::Insert, &["blah"][..]),
347    /// ]);
348    /// ```
349    ///
350    /// Due to lifetime restrictions it's currently impossible for the
351    /// returned slices to outlive the lookup.
352    pub fn iter_slices<'lookup, Old, New, T>(
353        &self,
354        old: &'lookup Old,
355        new: &'lookup New,
356    ) -> impl Iterator<Item = (ChangeTag, &'lookup T)> + use<'lookup, Old, New, T>
357    where
358        T: 'lookup + ?Sized,
359        Old: Index<Range<usize>, Output = T> + ?Sized,
360        New: Index<Range<usize>, Output = T> + ?Sized,
361    {
362        match *self {
363            DiffOp::Equal { old_index, len, .. } => {
364                Some((ChangeTag::Equal, &old[old_index..old_index + len]))
365                    .into_iter()
366                    .chain(None)
367            }
368            DiffOp::Insert {
369                new_index, new_len, ..
370            } => Some((ChangeTag::Insert, &new[new_index..new_index + new_len]))
371                .into_iter()
372                .chain(None),
373            DiffOp::Delete {
374                old_index, old_len, ..
375            } => Some((ChangeTag::Delete, &old[old_index..old_index + old_len]))
376                .into_iter()
377                .chain(None),
378            DiffOp::Replace {
379                old_index,
380                old_len,
381                new_index,
382                new_len,
383            } => Some((ChangeTag::Delete, &old[old_index..old_index + old_len]))
384                .into_iter()
385                .chain(Some((
386                    ChangeTag::Insert,
387                    &new[new_index..new_index + new_len],
388                ))),
389        }
390    }

392    pub(crate) fn is_empty(&self) -> bool {
393        let (_, old, new) = self.as_tag_tuple();
394        is_empty_range(&old) && is_empty_range(&new)
395    }

421    fn adjust(&mut self, adjust_offset: (usize, bool), adjust_len: (usize, bool)) {
422        #[inline(always)]
423        fn modify(val: &mut usize, adj: (usize, bool)) {
424            if adj.1 {
425                *val -= adj.0;
426            } else {
427                *val += adj.0;
428            }
429        }
430
431        match self {
432            DiffOp::Equal {
433                old_index,
434                new_index,
435                len,
436            } => {
437                modify(old_index, adjust_offset);
438                modify(new_index, adjust_offset);
439                modify(len, adjust_len);
440            }
441            DiffOp::Delete {
442                old_index,
443                old_len,
444                new_index,
445            } => {
446                modify(old_index, adjust_offset);
447                modify(old_len, adjust_len);
448                modify(new_index, adjust_offset);
449            }
450            DiffOp::Insert {
451                old_index,
452                new_index,
453                new_len,
454            } => {
455                modify(old_index, adjust_offset);
456                modify(new_index, adjust_offset);
457                modify(new_len, adjust_len);
458            }
459            DiffOp::Replace {
460                old_index,
461                old_len,
462                new_index,
463                new_len,
464            } => {
465                modify(old_index, adjust_offset);
466                modify(old_len, adjust_len);
467                modify(new_index, adjust_offset);
468                modify(new_len, adjust_len);
469            }
470        }
471    }

480    /// The text interface can produce changes over [`DiffableStr`] implementing
481    /// values.  As those are generic interfaces for different types of strings
482    /// utility methods to make working with standard rust strings more enjoyable.
483    impl<'s, T: DiffableStr + ?Sized> Change<&'s T> {
484        /// Returns the value as string if it is utf-8.
485        pub fn as_str(&self) -> Option<&'s str> {
486            T::as_str(self.value)
487        }
488
489        /// Returns the value (lossy) decoded as utf-8 string.
490        pub fn to_string_lossy(&self) -> Cow<'s, str> {
491            T::to_string_lossy(self.value)
492        }
493
494        /// Returns `true` if this change does not end in a newline and must be
495        /// followed up by one if line based diffs are used.
496        ///
497        /// The [`std::fmt::Display`] implementation of [`Change`] will automatically
498        /// insert a newline after the value if this is true.
499        pub fn missing_newline(&self) -> bool {
500            !T::ends_with_newline(self.value)
501        }
502    }
503
504    impl<T: DiffableStr + ?Sized> fmt::Display for Change<&T> {
505        fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
506            write!(
507                f,
508                "{}{}",
509                self.to_string_lossy(),
510                if self.missing_newline() { "\n" } else { "" }
511            )
512        }
513    }
