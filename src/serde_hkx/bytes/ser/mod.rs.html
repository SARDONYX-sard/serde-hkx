24/// To hkx binary file data.
25///
26/// # Errors
27/// - When information necessary for binary data conversion is missing.
28/// - When a write to the wrong write position is requested.
29///
30/// # NOTE
31/// - To reproduce the original hkx, it is necessary to arrange them in dependency order using `HavokSort::sort_for_bytes` or similar.
32/// - If you convert in the order of hkx => XML => hkx, you may not be able to restore the original hkx due to data loss during XML conversion.
33pub fn to_bytes<V>(value: &V, header: &HkxHeader) -> Result<Vec<u8>>
34where
35    V: Serialize + ClassNamesWriter,
36{
37    to_bytes_with_opt(
38        value,
39        header,
40        ByteSerializer {
41            is_little_endian: match header.endian {
42                0 => false, // big endian
43                1 => true,  // little endian
44                invalid => InvalidEndianSnafu { invalid }.fail()?,
45            },
46            is_x86: match header.pointer_size {
47                4 => true,
48                8 => false,
49                invalid => UnsupportedPtrSizeSnafu { invalid }.fail()?,
50            },
51            ..Default::default()
52        },
53    )
54}

56/// Serialize to bytes with custom `BytesSerializer` settings.
57///
58/// # Errors
59/// - When information necessary for binary data conversion is missing.
60/// - When a write to the wrong write position is requested.
61pub fn to_bytes_with_opt<V>(value: &V, header: &HkxHeader, ser: ByteSerializer) -> Result<Vec<u8>>
62where
63    V: Serialize + ClassNamesWriter,
64{
65    let mut serializer = ser;
66
67    // 1/5: root header
68    serializer.output.write_all(&header.to_bytes())?;
69    serializer.contents_class_name_section_index = header.contents_class_name_section_index;
70    serializer.contents_section_index = header.contents_section_index;
71
72    // 2/5: Get Section headers positions.(because the values of the fixups are not yet known.)
73    let classnames_header_start = 64 + header.padding_size() as u64; // 64: Root header size
74    let types_header_start = classnames_header_start + 48; // next `SectionHeader`(size 48bytes) position.
75    let data_header_start = types_header_start + 48;
76    serializer.output.set_position(data_header_start + 48);
77
78    // 3/5: section contents
79    // - `__classnames__` section
80    if serializer.is_little_endian {
81        serializer.class_starts =
82            value.write_classnames_section::<LittleEndian>(&mut serializer.output)?;
83    } else {
84        serializer.class_starts =
85            value.write_classnames_section::<BigEndian>(&mut serializer.output)?;
86    };
87    #[cfg(feature = "tracing")]
88    tracing::trace!("class_starts: {:#?}", serializer.class_starts);
89
90    // - `__data__` section
91    serializer.abs_data_offset = header.padding_size() + serializer.output.position() as u32;
92    value.serialize(&mut serializer)?;
93
94    // 4/5: Write fixups_offsets of `__data__` section header.
95    serializer.output.zero_fill_align(16)?; // Always make the start of fixups a multiple of 16.
96    let (local_offset, global_offset, virtual_offset) = serializer.write_data_fixups()?; // Write local, global and virtual fixups
97    let exports_offset = serializer.relative_position()?; // This is where the exports_offset is finally obtained.
98
99    // 5/5 Write remain offsets for `__types__` & `__data__` section header.
100    let abs_data_start = serializer.abs_data_offset;
101    let data_section_header = SectionHeader {
102        section_tag: SectionHeader::DATA_SECTION_HEADER_TAG,
103        section_tag_separator: 0xff,
104        absolute_data_start: abs_data_start,
105        local_fixups_offset: local_offset,
106        global_fixups_offset: global_offset,
107        virtual_fixups_offset: virtual_offset,
108        exports_offset,
109        imports_offset: exports_offset,
110        end_offset: exports_offset,
111    };
112    #[cfg(feature = "tracing")]
113    tracing::trace!("data_section_header = {data_section_header}");
114
115    serializer.output.set_position(classnames_header_start);
116    let section_offset = header.section_offset;
117
118    if serializer.is_little_endian {
119        tri!(serializer.write_section_headers::<LittleEndian>(
120            section_offset,
121            types_header_start,
122            data_header_start,
123            &data_section_header,
124        ));
125    } else {
126        tri!(serializer.write_section_headers::<BigEndian>(
127            section_offset,
128            types_header_start,
129            data_header_start,
130            &data_section_header,
131        ));
132    };
133
134    Ok(serializer.output.into_inner())
135}

137/// Binary data serializer
138///
139/// # Note
140/// - All of these fixups are from the `__data__` section.
141#[derive(Debug, Default)]
142pub struct ByteSerializer {
143    /// Endianness of serialization target
144    is_little_endian: bool,
145    /// Ptr size of serialization target.
146    /// 32bit: x86
147    is_x86: bool,
148
149    /// Bytes
150    output: Cursor<Vec<u8>>,
151
152    /// This is cached to find the relative position of the binary.
153    abs_data_offset: u32,
154
155    /// Flag to branch special align16 conditions such as the following.
156    /// Switch only in `SerializeStruct::serialize_array` and read only in `serialize_cow`.
157    /// - `hkArray<CString>` / `hkArray<StringPtr>` do not align16 each string.
158    /// - For a single field, align16 is done after string serialization.
159    is_in_str_array: bool,
160
161    // ---- local fixup information
162    /// The position to which the pointer returns after writing the end of the pointer.
163    /// Holds positional information for writing the destination of the nested array pointers in the field,
164    /// where the Nth element means N hierarchically nested.
165    /// - Example: `u32` <- `ClassB.b: u32` <- `ClassA.a: Array<ClassB>` <- `Array<ClassA>`
166    pointed_pos: Vec<u64>,
167    /// Coordination information to associate a pointer of a pointer type of a field in a class with the data location to which it points.
168    ///
169    /// # Note
170    /// All of these fixups are from the DATA SECTION.
171    ///
172    /// The following are not recorded in `local_fixup`.
173    /// - If `Array<T>` is empty.
174    /// - `CString`, `StringPtr` points to null ptr.
175    local_fixups: Vec<u8>,
176
177    // ---- Global fixup information
178    /// A map that holds the src of global_fixups until the dst of virtual_fixups is known.
179    /// -   key: Starting point of the binary for which the pointer class write is requested.
180    /// - value: Unique class pointer.(e.g. XML: #0050 -> 50)
181    global_fixups_ptr_src: IndexMap<u32, Pointer>,
182    /// The `global_fixup.dst` == `virtual_fixup.src`.
183    ///
184    /// Therefore, the write start position must be retained.
185    /// The position of dst of global_fixup will be known after all the binary data of all classes are written.
186    /// - key: Unique class pointer.(e.g. XML: #0050 -> 50)
187    /// - value: Starting point where Havok Class binary data is written.
188    ///
189    /// # Note
190    /// - This is used as a key since no duplicate ptr is required at the same relative position.
191    ///   The ptr may be shared-referenced and cannot be keyed.
192    virtual_fixups_ptr_src: HashMap<Pointer, u32>,
193    /// Index of the contents section.(To write `global_fixups` index)
194    ///
195    /// It is usually `2` and refers to the index of `__data__`.
196    contents_section_index: i32,
197
198    // ---- Virtual fixup information
199    /// C++ Class constructor positions map binary temporally buffer.
200    ///
201    /// Finally, write to the data for output.
202    virtual_fixups: Vec<u8>,
203    /// This information is needed in `virtual_fixup.name_offset`.
204    ///
205    /// This is created by writing the `__classnames__` section.
206    /// - key: class name
207    /// - value: class name start position
208    class_starts: ClassStartsMap,
209    /// Index of the contents class name section.(To write virtual fixups index)
210    ///
211    /// It is usually `0` and refers to the index of `__classnames__`.
212    contents_class_name_section_index: i32,
213}

215impl ByteSerializer {
216    /// Get the position relative to the start of the `__data__` section.
217    #[inline]
218    fn relative_position(&self) -> Result<u32> {
219        let position = self.output.position() as u32;
220        let abs_data_offset = self.abs_data_offset;
221        position
222            .checked_sub(abs_data_offset)
223            .ok_or(Error::OverflowSubtractAbs {
224                position,
225                abs_data_offset,
226            })
227    }
228
229    /// Write `global_fixups` of data section bytes to writer.
230    ///
231    /// # Info
232    /// If all virtual_fixups are not obtained, references may not be available?
233    ///
234    /// # Note
235    /// `global_fixup.dst` == `virtual_fixup.src`
236    fn write_global_fixups(&mut self) -> Result<()> {
237        for (&src, ptr) in &self.global_fixups_ptr_src {
238            if let Some(&dst) = self.virtual_fixups_ptr_src.get(ptr) {
239                #[cfg(feature = "tracing")]
240                {
241                    let src_abs = self.abs_data_offset + src;
242                    let dst_abs = self.abs_data_offset + dst;
243                    tracing::debug!(
244                        "[global_fixups]({:#x}) src({src}:{src:#x}/abs {src_abs:#x}) -> {ptr} dst({dst}:{dst:#x}/abs: {dst_abs:#x})",
245                        self.output.position()
246                    );
247                }
248
249                if self.is_little_endian {
250                    self.output.write_u32::<LittleEndian>(src)?; // src
251                    self.output
252                        .write_i32::<LittleEndian>(self.contents_section_index)?; // dst_section_index
253                    self.output.write_u32::<LittleEndian>(dst)?; // dst(virtual_fixup.dst)
254                } else {
255                    self.output.write_u32::<BigEndian>(src)?; // src
256                    self.output
257                        .write_i32::<BigEndian>(self.contents_section_index)?; // dst_section_index
258                    self.output.write_u32::<BigEndian>(dst)?; // dst(virtual_fixup.dst)
259                }
260            } else {
261                return MissingGlobalFixupClassSnafu { ptr: *ptr }.fail();
262            }
263        }
264        Ok(())
265    }
266
267    /// Write to temporary virtual_fixup data.
268    ///
269    /// # Info
270    /// Since the `class_name` and its location are known when the `__classnames__` section is written, the pair can be
271    /// written the moment virtual_fixup.src is available.
272    fn write_virtual_fixups_pair(
273        &mut self,
274        class_name: &'static str,
275        virtual_src: u32,
276    ) -> Result<()> {
277        if let Some(class_name_offset) = self.class_starts.get(class_name) {
278            if self.is_little_endian {
279                self.virtual_fixups.write_u32::<LittleEndian>(virtual_src)?; // src
280                self.virtual_fixups
281                    .write_i32::<LittleEndian>(self.contents_class_name_section_index)?; // dst_section_index, `__classnames__` section is 0
282                self.virtual_fixups
283                    .write_u32::<LittleEndian>(*class_name_offset)?; // dst(virtual_fixup.dst)
284            } else {
285                self.virtual_fixups.write_u32::<BigEndian>(virtual_src)?; // src
286                self.virtual_fixups
287                    .write_i32::<BigEndian>(self.contents_class_name_section_index)?; // dst_section_index, `__classnames__` section is 0
288                self.virtual_fixups
289                    .write_u32::<BigEndian>(*class_name_offset)?; // dst(virtual_fixup.dst)
290            };
291            Ok(())
292        } else {
293            MissingClassInClassnamesSectionSnafu { class_name }.fail()
294        }
295    }
296
297    /// Write all(`local`, `global` and `virtual`) fixups of data section.
298    ///
299    /// # Returns
300    /// (`local_offset`, `global_offset`, `virtual_offset`)
301    fn write_data_fixups(&mut self) -> Result<(u32, u32, u32)> {
302        let local_offset = self.relative_position()?;
303        self.output.write_all(&self.local_fixups)?;
304        self.output.align(16, 0xff)?;
305
306        #[cfg(feature = "tracing")]
307        tracing::debug!(
308            "[global_fixups pointers]\nsrc: {:#?},\ndest(same as virtual.src): {:#?}",
309            self.global_fixups_ptr_src,
310            self.virtual_fixups_ptr_src
311        );
312        let global_offset = self.relative_position()?;
313        self.write_global_fixups()?;
314        self.output.align(16, 0xff)?;
315
316        let virtual_offset = self.relative_position()?;
317        self.output.write_all(&self.virtual_fixups)?;
318        self.output.align(16, 0xff)?;
319        Ok((local_offset, global_offset, virtual_offset))
320    }
321
322    /// Write all section headers.
323    #[inline]
324    fn write_section_headers<B>(
325        &mut self,
326        section_offset: i16,
327        types_header_start: u64,
328        data_header_start: u64,
329        data_section_header: &SectionHeader,
330    ) -> Result<()>
331    where
332        B: ByteOrder,
333    {
334        // `__classnames__` header`
335        SectionHeader::write_classnames::<B>(
336            &mut self.output,
337            section_offset,
338            self.abs_data_offset,
339        )?;
340        // `__types__` header`
341        self.output.set_position(types_header_start);
342        SectionHeader::write_types::<B>(&mut self.output, self.abs_data_offset)?;
343        // `__data__` header`
344        self.output.set_position(data_header_start);
345        data_section_header.write_bytes::<B>(&mut self.output)?;
346        Ok(())
347    }
348
349    /// Go to relative The data position pointed to by ptr.
350    ///
351    /// Returns
352    /// (
353    ///  The last write position for the pointer type,
354    ///  `__data__` section relative offset) destination position,
355    /// )
356    ///
357    /// # Note
358    /// Always align to 16 bytes when writing to the array.
359    /// Note: If a Struct within hkArray<Struct> contains a StringPtr, it does not necessarily align to 16 bytes.
360    #[inline]
361    fn goto_latest_local_dst(&mut self, need_align16: bool) -> Result<(u64, u32)> {
362        let dest_abs_pos = tri!(
363            self.pointed_pos
364                .last()
365                .copied()
366                .context(NotFoundPointedPositionSnafu)
367        );
368        let dest_abs_pos = if need_align16 {
369            align!(dest_abs_pos, 16_u64)
370        } else {
371            dest_abs_pos
372        };
373        self.output.set_position(dest_abs_pos);
374        let local_dst = tri!(self.relative_position());
375        Ok((dest_abs_pos, local_dst))
376    }
377
378    fn update_last_local_dst(&mut self, new_pos: u64) -> Result<()> {
379        self.pointed_pos
380            .last_mut()
381            .map(|last| *last = new_pos)
382            .context(NotFoundPointedPositionSnafu)?; // Update to serialize the next pointed data.
383        Ok(())
384    }
385
386    /// Write a pair of local_fixups.
387    #[inline]
388    fn write_local_fixup_pair(&mut self, local_src: u32, local_dst: u32) -> Result<()> {
389        #[cfg(feature = "tracing")]
390        {
391            let src_abs = self.abs_data_offset + local_src;
392            let dst_abs = self.abs_data_offset + local_dst;
393            tracing::debug!(
394                "[local_fixup] src({local_src}/abs: {src_abs:#x}), dst({local_dst}/abs: {dst_abs:#x})"
395            );
396        }
397        if self.is_little_endian {
398            self.local_fixups.write_u32::<LittleEndian>(local_src)?;
399            self.local_fixups.write_u32::<LittleEndian>(local_dst)?;
400        } else {
401            self.local_fixups.write_u32::<BigEndian>(local_src)?;
402            self.local_fixups.write_u32::<BigEndian>(local_dst)?;
403        }
404        Ok(())
405    }
406
407    /// Write the internal data pointed to by the pointer of `CString` or `StringPtr`.
408    fn serialize_cow(&mut self, v: &Option<Cow<'_, str>>) -> Result<()> {
409        #[cfg(feature = "tracing")]
410        tracing::trace!("pointed_pos:({:#x?})", self.pointed_pos);
411
412        // Skip if `Option::None`(null pointer).
413        if let Some(v) = v {
414            let rel_ptr_start = self.relative_position()?;
415            tri!(self.serialize_ulong(Ulong::new(0))); // ptr size
416
417            let abs_next_ser_pos = self.output.position();
418            #[cfg(feature = "tracing")]
419            tracing::trace!("Serialize `CString`/`StringPtr` ({abs_next_ser_pos:#x}): (\"{v}\")",);
420
421            // local dst location -----------------------------------------------------
422            {
423                let (_, local_dst) = tri!(self.goto_latest_local_dst(false));
424                tri!(self.write_local_fixup_pair(rel_ptr_start, local_dst));
425            }
426
427            let c_string = StdCString::new(v.as_bytes())?;
428            let _ = self.output.write(c_string.as_bytes_with_nul())?;
429            self.output
430                .zero_fill_align(if self.is_in_str_array { 2 } else { 16 })?;
431
432            tri!(self.update_last_local_dst(self.output.position()));
433            // local dst --------------------------------------------------------------
434
435            self.output.set_position(abs_next_ser_pos); // go back to C++ field serialization position.
436        } else {
437            tri!(self.serialize_ulong(Ulong::new(0))); // null ptr, just write ptr size
438        }
439
440        Ok(())
441    }
442}

444/// Endianness and a common write process that takes into account whether the array is being serialized or not.
445macro_rules! impl_serialize_primitive {
446    ($method:ident, $value_type:ty, $write:ident) => {
447        #[inline]
448        fn $method(self, v: $value_type) -> Result<Self::Ok, Self::Error> {
449            match self.is_little_endian {
450                true => self.output.$write::<LittleEndian>(v),
451                false => self.output.$write::<BigEndian>(v),
452            }?;
453            Ok(())
454        }
455    };
456}

458/// Endianness and a common write process that takes into account whether the array is being serialized or not.
459macro_rules! impl_serialize_math {
460    ($method:ident, $value_type:ty) => {
461        fn $method(self, v: &$value_type) -> Result<Self::Ok, Self::Error> {
462            match self.is_little_endian {
463                true => self.output.write(v.to_le_bytes().as_slice()),
464                false => self.output.write(v.to_be_bytes().as_slice()),
465            }?;
466            Ok(())
467        }
468    };

526    // Register data to be written to global_fixups.
527    // Until the data in virtual_fixups is determined (until all C++ classes are written), temporarily write to `IndexMap` as the value of dst is unknown.
528    fn serialize_pointer(self, ptr: Pointer) -> Result<Self::Ok, Self::Error> {
529        #[allow(clippy::needless_else)]
530        if !ptr.is_null() {
531            // Write global_fixup src(write start) position.
532            let start = self.relative_position()?;
533            #[cfg(feature = "tracing")]
534            tracing::debug!(
535                "Insert `global_fixup.src`({start}:{start:#x}/abs {:#x}): {ptr}",
536                self.output.position()
537            );
538            self.global_fixups_ptr_src.insert(start, ptr);
539        } else {
540            #[cfg(feature = "tracing")]
541            tracing::debug!("Skip global_fixup.src writing, because it's null ptr.");
542        };
543        self.serialize_ulong(Ulong::new(0))
544    }

551    /// This is called in the Havok Class array or HashMap, or in a serializer in a field.
552    /// Classes in the field may be inlined, in which case `class_meta` will be [`None`].
553    ///
554    /// # what's `class_meta`?
555    /// If `class_meta` exists, it is when writing an `hkobject` with the `name` attribute in XML.
556    /// This must be written to virtual_fixup so that the constructor can be called.
557    fn serialize_struct(
558        self,
559        name: &'static str,
560        class_meta: Option<(Pointer, Signature)>,
561        sizes: (u64, u64),
562    ) -> Result<Self::SerializeStruct, Self::Error> {
563        #[allow(clippy::needless_else)]
564        let is_root = if let Some((ptr, _sig)) = class_meta {
565            self.output.zero_fill_align(16)?; // Make sure `virtual_fixup.src`(each Class) is `align16`.
566
567            let virtual_fixup_abs = self.output.position();
568            #[cfg(feature = "tracing")]
569            tracing::debug!(
570                "serialize struct {name}(index = {ptr}, signature = {_sig}, abs_position = {virtual_fixup_abs:#x})"
571            );
572            {
573                let virtual_src = self.relative_position()?;
574                self.write_virtual_fixups_pair(name, virtual_src)?; // Ok, `virtual_fixup` is known.
575                self.virtual_fixups_ptr_src.insert(ptr, virtual_src); // Backup to write `global_fixups`
576            }
577
578            // The data pointed to by the pointer member(hkCString/hkStringPtr/`hkArray<T>`)
579            // must first be aligned 16 bytes before it is written.
580            let size = if self.is_x86 { sizes.0 } else { sizes.1 };
581            let abs_latest_local_dst = align!(virtual_fixup_abs + size, 16_u64);
582            self.pointed_pos.push(abs_latest_local_dst);
583            true
584        } else {
585            #[cfg(feature = "tracing")]
586            tracing::debug!("serialize struct {name}(A class within a field.)");
587            false
588        };
589
590        Ok(Self::SerializeStruct::new(self, is_root))
591    }

618    fn serialize_half(self, v: f16) -> Result<Self::Ok, Self::Error> {
619        let bytes = if self.is_little_endian {
620            v.to_le_bytes()
621        } else {
622            v.to_be_bytes()
623        };
624        self.output.write_all(&bytes)?;
625        Ok(())
626    }
