17/// Parses the start tag `<hksection name="__data__">`
18///
19/// # Errors
20/// When parse failed.
21pub fn hksection_start_tag(input: &mut &str) -> ModalResult<()> {
22    seq!(
23        _: delimited_comment_multispace0("<"),
24        _: delimited_multispace0(Caseless("hksection")),
25        _: Caseless("name"),
26        _: delimited_multispace0("="),
27        _: "\"__data__\"",
28        _: delimited_multispace0_comment(">"),
29    )
30    .context(StrContext::Label("hksection start tag"))
31    .context(StrContext::Label("<hksection name=\"__data__\">"))
32    .parse_next(input)?;
33    Ok(())
34}

36/// Parses the start tag `<tag>`
37pub fn start_tag<'a>(tag: &'static str) -> impl Parser<&'a str, (), ErrMode<ContextError>> {
38    seq!(
39        _: delimited_comment_multispace0("<"),
40        _: delimited_multispace0(Caseless(tag)),
41        _: delimited_multispace0_comment(">")
42    )
43    .context(StrContext::Label("start tag"))
44    .context(StrContext::Label(tag))
45}
46
47/// Parses the end tag `</tag>`
48pub fn end_tag<'a>(tag: &'static str) -> impl Parser<&'a str, (), ErrMode<ContextError>> {
49    seq!(
50        _: delimited_comment_multispace0("<"),
51        _: delimited_multispace0("/"),
52        _: delimited_multispace0(Caseless(tag)),
53        _: delimited_multispace0_comment(">")
54    )
55    .context(StrContext::Label("end tag"))
56    .context(StrContext::Label(tag))
57}

59/// Parses the array start tag (e.g. `<hkobject name="#0010" class="hkbProjectData" signature="0x13a39ba7">`)
60///
61/// # Returns
62/// ([`Pointer`], ClassName, [`Signature`]) -> e.g. (`#0010`, `"hkbProjectData"`, `0x13a39ba7`)
63///
64/// # Errors
65/// When parse failed.
66pub fn class_start_tag<'a>(input: &mut &'a str) -> ModalResult<(Pointer, &'a str, Signature)> {
67    seq!(
68        _: delimited_comment_multispace0("<"),
69        _: delimited_multispace0("hkobject"),
70        _: delimited_multispace0("name"),
71        _: delimited_multispace0("="),
72        attr_ptr,
73
74        _: delimited_multispace0("class"),
75        _: delimited_multispace0("="),
76        attr_string,
77
78        _: delimited_multispace0("signature"),
79        _: delimited_multispace0("="),
80        _: delimited_multispace0("\""),
81        radix_digits.map(|digits| Signature::new(digits as u32)),
82        _: delimited_multispace0("\""),
83        _: delimited_multispace0_comment(">")
84    )
85    .context(StrContext::Label("Class start tag"))
86    .context(StrContext::Expected(StrContextValue::Description(
87        r##"e.g. `<hkobject name="#0010" class="hkbProjectData" signature="0x13a39ba7">`"##,
88    )))
89    .parse_next(input)
90}

92/// Parses the field of class start opening tag `<hkparam name=`
93///
94/// # Note
95/// All arguments are used only for clarity of error reporting.
96pub fn field_start_open_tag<'a>(
97    class_name: &'static str,
98) -> impl Parser<&'a str, (), ErrMode<ContextError>> {
99    seq!(
100        _: delimited_comment_multispace0("<"),
101        _: delimited_multispace0("hkparam"),
102        _: delimited_multispace0("name"),
103        _: delimited_multispace0("="),
104    )
105    .context(StrContext::Label("field of class: start opening tag"))
106    .context(StrContext::Label(class_name))
107    .context(StrContext::Expected(StrContextValue::Description(
108        "e.g. `<hkparam name=`",
109    )))
110}

112/// Parses the field of class start closing tag `>`, `numelements="0">`, `/>`, or `numelements="0" />`
113///
114/// # Returns
115/// (numelements, is_self_closing)
116///
117/// # Errors
118/// When parse failed.
119pub fn field_start_close_tag(input: &mut &str) -> ModalResult<(Option<u64>, bool)> {
120    use winnow::combinator::alt;
121
122    alt((
123        // Handle self-closing tag with numelements: numelements="0" />
124        seq!(
125            seq!(
126                _: delimited_multispace0("numelements"),
127                _: delimited_multispace0("="),
128                number_in_string::<u64>, // e.g. "8"
129            ),
130            _: delimited_multispace0("/"),
131            _: delimited_multispace0_comment(">")
132        )
133        .map(|((n,),)| (Some(n), true)),
134        //
135        // Handle regular closing tag with numelements: numelements="0">
136        seq!(
137            seq!(
138                _: delimited_multispace0("numelements"),
139                _: delimited_multispace0("="),
140                number_in_string::<u64>, // e.g. "8"
141            ),
142            _: delimited_multispace0_comment(">")
143        )
144        .map(|((n,),)| (Some(n), false)),
145        //
146        // Handle self-closing tag without numelements: />
147        seq!(
148            _: delimited_multispace0("/"),
149            _: delimited_multispace0_comment(">")
150        )
151        .map(|()| (None, true)),
152        //
153        // Handle regular closing tag without numelements: >
154        seq!(
155            _: delimited_multispace0_comment(">")
156        )
157        .map(|()| (None, false)),
158    ))
159    .context(StrContext::Label("field of class: start closing tag"))
160    .context(StrContext::Expected(StrContextValue::Description(
161        "e.g. `>`, `/>`, `numelements=\"0\">`, or `numelements=\"0\" />`",
162    )))
163    .parse_next(input)
164}

170/// Parses a number inside a string, e.g., `"64"`
171///
172/// # Errors
173/// When parse failed.
174pub fn number_in_string<Num>(input: &mut &str) -> ModalResult<Num>
175where
176    Num: FromStr,
177{
178    attr_string
179        .parse_to()
180        .context(StrContext::Label("number in string"))
181        .context(StrContext::Expected(Description(r#"Number(e.g. `"64"`)"#)))
182        .parse_next(input)
183}

185/// Parses a xml attribute string(surrounded double quotes), e.g. `"string"`
186///
187/// # Errors
188/// When parse failed.
189pub fn attr_string<'a>(input: &mut &'a str) -> ModalResult<&'a str> {
190    delimited("\"", take_until(0.., "\""), "\"")
191        .context(StrContext::Label("String in XML attribute"))
192        .context(StrContext::Expected(Description(r#"String(e.g. `"Str"`)"#)))
193        .parse_next(input)
194}

196/// Parser a xml attribute pointer in string, e.g. `"#0050"`
197///
198/// # Errors
199/// When parse failed.
200fn attr_ptr(input: &mut &str) -> ModalResult<Pointer> {
201    delimited("\"", pointer, "\"").parse_next(input)
202}

204/// Parse radix digits. e.g. `0b101`, `0xff`
205///
206/// # Errors
207/// When parse failed.
208fn radix_digits(input: &mut &str) -> ModalResult<usize> {
209    dispatch!(take(2_usize);
210        "0b" | "0B" => digit1.try_map(|s| usize::from_str_radix(s, 2))
211                        .context(StrContext::Label("digit")).context(StrContext::Expected(StrContextValue::Description("binary"))),
212        "0o" | "0O" => oct_digit1.try_map(|s| usize::from_str_radix(s, 8))
213                        .context(StrContext::Label("digit")).context(StrContext::Expected(StrContextValue::Description("octal"))),
214        "0d" | "0D" => digit1.try_map(|s: &str| s.parse::<usize>())
215                        .context(StrContext::Label("digit")).context(StrContext::Expected(StrContextValue::Description("decimal"))),
216        "0x" | "0X" => hex_digit1.try_map(|s|usize::from_str_radix(s, 16))
217                        .context(StrContext::Label("digit")).context(StrContext::Expected(StrContextValue::Description("hexadecimal"))),
218        _ => fail.context(StrContext::Label("radix prefix"))
219                .context(StrContext::Expected(StrContextValue::StringLiteral("0b")))
220                .context(StrContext::Expected(StrContextValue::StringLiteral("0o")))
221                .context(StrContext::Expected(StrContextValue::StringLiteral("0d")))
222                .context(StrContext::Expected(StrContextValue::StringLiteral("0x"))),
223    ).parse_next(input)
224}
