1//! Deserializing each field element in an `Struct`
2use super::{XmlDeserializer, parser::tag::class_start_tag};
3use crate::{errors::de::Error, tri};
4use havok_serde::de::{ClassIndexAccess, DeserializeSeed};
5
6/// A structure for deserializing each element in an `Struct`.
7///
8/// # Expected XML
9/// ```xml
10/// <hkobject name="#0010" class="hkbProjectData" signature="0x13a39ba7"> <!-- <-Parsed by `deserialize_struct` -->
11///   <!-- memSizeAndFlags SERIALIZE_IGNORED -->
12///   <!-- referenceCount SERIALIZE_IGNORED -->
13///   <hkparam name="worldUpWS">(0.000000 0.000000 1.000000 0.000000)</hkparam>
14///   <hkparam name="stringData">#0009</hkparam>
15///   <hkparam name="defaultEventMode">EVENT_MODE_IGNORE_FROM_GENERATOR</hkparam>
16/// </hkobject>
17/// ```
18#[derive(Debug)]
19pub struct ClassIndexMapDeserializer<'a, 'de: 'a> {
20    /// Deserializer
21    de: &'a mut XmlDeserializer<'de>,
22}
23
24impl<'a, 'de> ClassIndexMapDeserializer<'a, 'de> {
25    /// Create a new map deserializer
26    #[inline]
27    pub const fn new(de: &'a mut XmlDeserializer<'de>) -> Self {
28        Self { de }
29    }
30}
31
32impl<'de> ClassIndexAccess<'de> for ClassIndexMapDeserializer<'_, 'de> {
33    type Error = Error;
34
35    fn next_key(&mut self) -> Result<&'de str, Self::Error> {
36        let (_ptr, class_name, _signature) = tri!(self.de.parse_peek(class_start_tag));
37        Ok(class_name)
38    }
39
40    #[inline]
41    fn next_value_seed<V>(&mut self, seed: V) -> Result<V::Value, Self::Error>
42    where
43        V: DeserializeSeed<'de>,
44    {
45        let value = seed.deserialize(&mut *self.de);
46        self.de.in_struct = false;
47        value
48    }
49}
