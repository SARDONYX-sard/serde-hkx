1//! An argument in a `format!`-like macro, parsable and transformable to tokens.
2
3use proc_macro2::TokenStream as TokenStream2;
4use quote::ToTokens;
5use syn::{
6    parse::{Parse, ParseStream, Result},
7    token, Expr, Ident, Token,
8};
9
10/// An argument in a `format!`-like macro (excluding the first argument aka the format string).
11pub struct FormatArg {
12    /// The argument name in the case of a named argument, e.g. `foo` inside `foo = 1 + 1`.
13    pub arg_name: Option<(Ident, token::Eq)>,
14    /// The real argument to be formatted by the macro.
15    pub expr: Expr,
16}
17
18impl Parse for FormatArg {
19    fn parse(input: ParseStream) -> Result<Self> {
20        let arg_name: Option<(Ident, token::Eq)> = if input.peek2(Token![=]) {
21            Some((input.parse()?, input.parse()?))
22        } else {
23            None
24        };
25        let expr: Expr = input.parse()?;
26
27        Ok(FormatArg { arg_name, expr })
28    }
29}
30
31impl ToTokens for FormatArg {
32    fn to_tokens(&self, tokens: &mut TokenStream2) {
33        if let Some((arg_name, eq)) = &self.arg_name {
34            arg_name.to_tokens(tokens);
35            eq.to_tokens(tokens);
36        }
37        self.expr.to_tokens(tokens);
38    }
39}
