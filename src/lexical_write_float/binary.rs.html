19/// Optimized float-to-string algorithm for power of 2 radixes.
20///
21/// This assumes the float is:
22///     1). Non-special (NaN or Infinite).
23///     2). Non-negative.
24///
25/// # Safety
26///
27/// Safe as long as `bytes` is large enough to hold the number of
28/// significant digits, any (optional) leading or trailing zeros,
29/// and the scientific exponent.
30///
31/// # Panics
32///
33/// Panics if exponent notation is used, and the exponent base and
34/// mantissa radix are not the same in `FORMAT`.
35#[inline]
36pub fn write_float<F: Float, const FORMAT: u128>(
37    float: F,
38    bytes: &mut [u8],
39    options: &Options,
40) -> usize
41where
42    <F as Float>::Unsigned: WriteInteger + FormattedSize,
43{
44    // PRECONDITIONS
45
46    // Assert no special cases remain, no negative numbers,
47    // and a valid format.
48    let format = NumberFormat::<{ FORMAT }> {};
49    assert!(format.is_valid());
50    debug_assert!(!float.is_special());
51    debug_assert!(float >= F::ZERO);
52
53    // Quickly calculate the number of bits we would have written.
54    // This simulates writing the digits, so we can calculate the
55    // scientific exponent. Since this number is often constant
56    // (except for denormal values), and doesn't describe
57    // the actual shift or digits we use...
58    //
59    // Note:
60    //      Except for denormal floats, this will always
61    //      be `F::MANTISSA_SIZE`, unless we have special
62    //      formatting write control.
63    let mantissa = float.mantissa();
64    let radix = format.mantissa_radix();
65    let (mantissa, mantissa_bits) = truncate_and_round(mantissa, radix, options);
66
67    // See if we should use an exponent if the number was represented
68    // in scientific notation, AKA, `I.FFFF^EEE`. If the exponent is above
69    // a certain value, then use scientific notation. We therefore have
70    // to adjust the exponent by the number of mantissa bits, and shift
71    // by 1 (since a scientific exponent of 0 should have 1 digit ahead).
72    // This is a binary exp, so we need to how large our
73    // adjusted exp to the radix is.
74    //
75    // The scientific exponent is always this way: it's the float exponent
76    // (binary) + mantissa bits (binary) - 1 (for the first bit, binary),
77    // since 1.0 is scientific exponent 0. We need to check the scientific
78    // exponent relative to the number of leading or trailing 0s
79    // it would introduce, that is, scaled to bits/digit. The min exp must
80    // be less than, and the max must be above 0.
81    let exp = float.exponent();
82    let mut sci_exp = exp + mantissa_bits as i32 - 1;
83
84    // Normalize the exponent if we have an actual zero.
85    if mantissa == <F as Float>::Unsigned::ZERO {
86        sci_exp = 0;
87    }
88
89    write_float!(
90        float,
91        FORMAT,
92        sci_exp,
93        options,
94        write_float_scientific,
95        write_float_positive_exponent,
96        write_float_negative_exponent,
97        generic => _,
98        bytes => bytes,
99        args => mantissa, exp, sci_exp, options,
100    )
101}

103/// Write float to string in scientific notation.
104///
105/// # Preconditions
106///
107/// The mantissa must be truncated and rounded, prior to calling this,
108/// based on the number of maximum digits. In addition, `exponent_base`
109/// and `mantissa_radix` in `FORMAT` must be identical.
110#[cfg_attr(not(feature = "compact"), inline(always))]
111pub fn write_float_scientific<M, const FORMAT: u128>(
112    bytes: &mut [u8],
113    mantissa: M,
114    exp: i32,
115    sci_exp: i32,
116    options: &Options,
117) -> usize
118where
119    M: WriteInteger + FormattedSize,
120{
121    // Just decent size bounds checks to ensure we have a lot of space.
122    assert!(M::FORMATTED_SIZE < BUFFER_SIZE - 2);
123    debug_assert!(bytes.len() >= BUFFER_SIZE);
124
125    // Config options
126    let format = NumberFormat::<{ FORMAT }> {};
127    let bits_per_digit = fast_log2(format.mantissa_radix());
128    let decimal_point = options.decimal_point();
129
130    // Validate our options: we don't support different exponent bases here.
131    debug_assert!(format.mantissa_radix() == format.exponent_base());
132
133    // Write our value, then trim trailing zeros, before we check the exact
134    // bounds of the digits, to avoid accidentally choosing too many digits.
135    let shl = calculate_shl(exp, bits_per_digit);
136    let value = mantissa << shl;
137
138    let count = value.write_mantissa::<FORMAT>(&mut bytes[1..]);
139    bytes[0] = bytes[1];
140    bytes[1] = decimal_point;
141    let zeros = rtrim_char_count(&bytes[2..count + 1], b'0');
142    let digit_count = count - zeros;
143    // Extra 1 since we have the decimal point.
144    let mut cursor = digit_count + 1;
145
146    // Determine if we need to add more trailing zeros.
147    let exact_count = shared::min_exact_digits(digit_count, options);
148
149    // Write any trailing digits to the output.
150    if !format.no_exponent_without_fraction() && cursor == 2 && options.trim_floats() {
151        // Need to trim floats from trailing zeros, and we have only a decimal.
152        cursor -= 1;
153    } else if exact_count < 2 {
154        // Need to have at least 1 digit, the trailing `.0`.
155        bytes[cursor] = b'0';
156        cursor += 1;
157    } else if exact_count > digit_count {
158        // NOTE: Neither `exact_count >= digit_count >= 2`.
159        // We need to write `exact_count - (cursor - 1)` digits, since
160        // cursor includes the decimal point.
161        let digits_end = exact_count + 1;
162        bytes[cursor..digits_end].fill(b'0');
163        cursor = digits_end;
164    }
165
166    // Now, write our scientific notation.
167    let scaled_sci_exp = scale_sci_exp(sci_exp, bits_per_digit);
168    shared::write_exponent::<FORMAT>(bytes, &mut cursor, scaled_sci_exp, options.exponent());
169
170    cursor
171}

173/// Write negative float to string without scientific notation.
174///
175/// Has a negative exponent (shift right) and no scientific notation.
176#[cfg_attr(not(feature = "compact"), inline(always))]
177pub fn write_float_negative_exponent<M, const FORMAT: u128>(
178    bytes: &mut [u8],
179    mantissa: M,
180    exp: i32,
181    sci_exp: i32,
182    options: &Options,
183) -> usize
184where
185    M: WriteInteger + FormattedSize,
186{
187    // NOTE:
188    //  This cannot trim trailing zeros, since the exponent **must**
189    //  be less than 0 and the value cannot be zero.
190    debug_assert!(mantissa != M::ZERO);
191    debug_assert!(sci_exp < 0);
192
193    // Just decent size bounds checks to ensure we have a lot of space.
194    assert!(M::FORMATTED_SIZE < BUFFER_SIZE - 2);
195    debug_assert!(bytes.len() >= BUFFER_SIZE);
196
197    // Config options
198    let format = NumberFormat::<{ FORMAT }> {};
199    let bits_per_digit = fast_log2(format.mantissa_radix());
200    let decimal_point = options.decimal_point();
201
202    // The number of 0 bits we need to pad left (reducing the
203    // exponent) is just the negative scientific exponent.
204    // We then need to calculate the number of zero digits
205    // from this, remembering that we're padding left,
206    // so for example, `1/2` in hex is represented as `0.8`.
207    // That means we need the `⌈ zero_bits / bits_per_digit ⌉`.
208    let zero_bits = sci_exp.wrapping_neg();
209    let zero_digits = fast_ceildiv(zero_bits, bits_per_digit) as usize;
210
211    // Write our 0 digits.
212    bytes[0] = b'0';
213    bytes[1] = decimal_point;
214    bytes[2..zero_digits + 1].fill(b'0');
215    let mut cursor = zero_digits + 1;
216
217    // Generate our digits after the shift. Store the number of written
218    // digits, so we can adjust the end-point accordingly.
219    let shl = calculate_shl(exp, bits_per_digit);
220    let value = mantissa << shl;
221
222    // Won't panic, if the buffer is large enough to hold the significant
223    // digits.
224    let count = value.write_mantissa::<FORMAT>(&mut bytes[cursor..]);
225    let zeros = rtrim_char_count(&bytes[cursor..cursor + count], b'0');
226    let digit_count = count - zeros;
227    cursor += digit_count;
228
229    // Determine if we need to add more trailing zeros.
230    let exact_count = shared::min_exact_digits(digit_count, options);
231
232    // Check if we need to write more trailing digits.
233    // NOTE: we cannot have a "0.1" case here, since we've previous truncated
234    // the significant digits, and the result is < 1.
235    if exact_count > digit_count {
236        let zeros = exact_count - digit_count;
237        bytes[cursor..cursor + zeros].fill(b'0');
238        cursor += zeros;
239    }
240
241    cursor
242}

244/// Write positive float to string without scientific notation.
245///
246/// Has a positive exponent (shift left) and no scientific notation.
247#[cfg_attr(not(feature = "compact"), inline(always))]
248pub fn write_float_positive_exponent<M, const FORMAT: u128>(
249    bytes: &mut [u8],
250    mantissa: M,
251    exp: i32,
252    sci_exp: i32,
253    options: &Options,
254) -> usize
255where
256    M: WriteInteger + FormattedSize,
257{
258    debug_assert!(sci_exp >= 0 || mantissa == M::ZERO);
259
260    // Config options
261    let format = NumberFormat::<{ FORMAT }> {};
262    let bits_per_digit = fast_log2(format.mantissa_radix());
263    let decimal_point = options.decimal_point();
264
265    // Write our value, then trim trailing zeros, before we check the exact
266    // bounds of the digits, to avoid accidentally choosing too many digits.
267    let shl = calculate_shl(exp, bits_per_digit);
268    let value = mantissa << shl;
269
270    let count = value.write_mantissa::<FORMAT>(bytes);
271    let zeros = rtrim_char_count(&bytes[..count], b'0');
272    let mut digit_count = count - zeros;
273
274    // Write the significant digits.
275    // Calculate the number of digits we can write left of the decimal point.
276    // If we have a scientific exp of 0, we still need
277    // to write 1 digit before, so it's `⌊ leading_bits / bits_per_digit ⌋ + 1`.
278    let leading_bits = sci_exp;
279    let leading_digits = (leading_bits / bits_per_digit) as usize + 1;
280
281    // Now need to write our decimal point and add any additional significant
282    // digits.
283    let mut cursor: usize;
284    let mut trimmed = false;
285    if leading_digits >= digit_count {
286        // We have more leading digits than digits we wrote: can write
287        // any additional digits, and then just write the remaining zeros.
288        bytes[digit_count..leading_digits].fill(b'0');
289        cursor = leading_digits;
290        // Only write decimal point if we're not trimming floats.
291        if !options.trim_floats() {
292            bytes[cursor] = decimal_point;
293            cursor += 1;
294            bytes[cursor] = b'0';
295            cursor += 1;
296            digit_count += 1;
297        } else {
298            trimmed = true;
299        }
300    } else {
301        // We have less leading digits than digits we wrote: find the
302        // decimal point index, shift all digits right by 1, then write it.
303        let shifted = digit_count - leading_digits;
304        let buf = &mut bytes[leading_digits..digit_count + 1];
305        for i in (0..shifted).rev() {
306            buf[i + 1] = buf[i];
307        }
308        bytes[leading_digits] = decimal_point;
309        cursor = digit_count + 1;
310    }
311
312    // Determine if we need to add more trailing zeros after a decimal point.
313    let exact_count = shared::min_exact_digits(digit_count, options);
314
315    // Change the number of digits written, if we need to add more or trim digits.
316    if !trimmed && exact_count > digit_count {
317        // Check if we need to write more trailing digits.
318        let zeros = exact_count - digit_count;
319        bytes[cursor..cursor + zeros].fill(b'0');
320        cursor += zeros;
321    }
322
323    cursor
324}

326// ALGORITHM
327// ---------
328
329// PARSER
330//
331// The simple, parser algorithm can be thought of like this:
332// ```python
333// import fractions
334// import numpy as np
335//
336// def parse_binary(integer, fraction, exponent, radix, dtype):
337//     '''Parse a binary (or power of 2) integer to float'''
338//
339//     # Parse our simple values
340//     iint = int(integer, radix)
341//     ifrac = int(fraction, radix)
342//     iexp = int(exponent, radix)
343//
344//     # Calculate our actual values
345//     fint_num = iint * 2**iexp
346//     fint = fractions.Fraction(fint_num)
347//
348//     if len(fraction) > iexp:
349//         ffrac_exp_num = 0
350//         ffrac_exp_den = len(fraction) - iexp
351//     else:
352//         ffrac_exp_num = iexp - len(fraction)
353//         ffrac_exp_den = 0
354//     ffrac_num = ifrac * 2**ffrac_exp_num
355//     ffrac_den = 2**ffrac_exp_den
356//     ffrac = fractions.Fraction(ffrac_num, ffrac_den)
357//
358//     return dtype(fint + ffrac)
359//
360// parse_binary('1', '001111000000110010', '0', 2, np.float32)
361// ```
362//
363// For binary floats, we can get a step further, assuming the value
364// is in the proper range for the exponent.
365//
366// Please note that the real implementation is much faster,
367// but for simple floats, this suffices.
368//
369// A another example is as follows. This closely tracks our internal logic
370// for converting extended-precision floats to native floats and creates
371// an exact result.
372//
373// ```python
374// import sys
375// import numpy as np
376//
377//
378// def into_float_bits(mant, exp, finfo):
379//     '''Converts a mantissa and exponent (binary) into a float.'''
380//      # Export floating-point number.
381//     if mant == 0 or exp < finfo.DENORMAL_EXPONENT:
382//         # sub-denormal, underflow
383//         return 0
384//     elif exp >= finfo.MAX_EXPONENT:
385//         return INFINITY_BITS
386//     else:
387//         if exp == finfo.DENORMAL_EXPONENT and mant & finfo.HIDDEN_BIT_MASK == 0:
388//             exp = 0
389//         else:
390//             exp += finfo.EXPONENT_BIAS
391//         exp <<= finfo.MANTISSA_SIZE
392//         mant &= finfo.MANTISSA_MASK
393//         return mant | exp
394//
395//
396// def into_float(mant, exp, finfo, is_positive):
397//     '''Converts a mantissa, exponent, and sign into a float.'''
398//
399//     bits = into_float_bits(mant, exp, finfo)
400//     if not is_positive:
401//         bits |= SIGN_MASK
402//
403//     as_bytes = bits.to_bytes(finfo.BITS // 8, sys.byteorder)
404//     return np.frombuffer(as_bytes, dtype=finfo.ftype)[0]
405//
406//
407// def ctlz(integer, finfo):
408//     '''Count the leading zeros on an integer'''
409//
410//     bits = [0] * finfo.BITS
411//     for bit in range(0, finfo.BITS):
412//         bits[bit] = (integer >> bit) & 1
413//
414//     count = 0
415//     bit = 31
416//     while bit >= 0 and bits[bit] == 0:
417//         count += 1
418//         bit -= 1
419//
420//     return count
421//
422//
423// def normalize(mant, exp, finfo):
424//     '''Normalizes the float.'''
425//
426//     if mant == 0:
427//         shift = 0
428//     else:
429//         shift = ctlz(mant, finfo)
430//
431//     mant <<= shift
432//     exp -= shift
433//
434//     return mant, exp
435//
436//
437// def lower_n_mask(n):
438//     '''Generate a bitwise mask for the lower `n` bits.'''
439//     return (1 << n) - 1
440//
441//
442// def nth_bit(n):
443//     '''Calculate a scalar factor of 2 above the halfway point.'''
444//     return 1 << n
445//
446//
447// def lower_n_halfway(n):
448//     '''Calculate the halfway point for the lower `n` bits.'''
449//
450//     if n == 0:
451//         return 0
452//     return nth_bit(n - 1)
453//
454//
455// def internal_n_mask(bit, n):
456//     '''Calculate a bitwise mask with `n` 1 bits starting at the `bit` position.'''
457//     return lower_n_mask(bit) ^ lower_n_mask(bit - n)
458//
459//
460// def round_nearest(mant, exp, finfo, shift):
461//     '''Shift right N-bytes and round to the nearest.'''
462//
463//     mask = lower_n_mask(shift)
464//     halfway = lower_n_halfway(shift)
465//
466//     truncated_bits = mant & mask;
467//     is_above = truncated_bits > halfway;
468//     is_halfway = truncated_bits == halfway;
469//
470//     if shift == finfo.BITS:
471//         mant = 0
472//     else:
473//         mant >>= shift
474//     exp += shift
475//
476//     return (mant, exp, is_above, is_halfway)
477//
478//
479// def tie_even(mant, exp, is_above, is_halfway):
480//     '''Tie rounded floating point to even.'''
481//
482//     is_odd = mant & 0x1 == 0x1
483//     if is_above or (is_odd and is_halfway):
484//         mant += 1
485//     return mant, exp
486//
487//
488// def round_nearest_tie_even(mant, exp, finfo, shift):
489//     '''Shift right N-bytes and round nearest, tie-to-even.'''
490//
491//     mant, exp, is_above, is_halfway = round_nearest(mant, exp, finfo, shift)
492//     return tie_even(mant, exp, is_above, is_halfway)
493//
494//
495// def round_to_float(mant, exp, finfo):
496//     '''Round the float to native, using round-nearest, tie-even.'''
497//
498//     final_exp = exp + finfo.DEFAULT_SHIFT
499//     if final_exp < finfo.DENORMAL_EXPONENT:
500//         diff = finfo.DENORMAL_EXPONENT - exp
501//         if diff < finfo.BITS:
502//             mant, exp = round_nearest_tie_even(mant, exp, finfo, diff)
503//         else:
504//             return (0, 0)
505//     else:
506//         mant, exp = round_nearest_tie_even(mant, exp, finfo, finfo.DEFAULT_SHIFT)
507//
508//     if mant & finfo.CARRY_MASK == finfo.CARRY_MASK:
509//         mant >>= 1
510//         exp += 1
511//
512//     return mant, exp
513//
514//
515// def avoid_overflow(mant, exp, finfo):
516//     '''Avoid overflow for large values, shift left as needed.'''
517//
518//     if exp > finfo.MAX_EXPONENT:
519//         diff = exp - MAX_EXPONENT
520//         if diff < finfo.MANTISSA_SIZE:
521//             bit = MANTISSA_SIZE + 1
522//             n = diff + 1
523//             mask = internal_n_mask(bit, n)
524//             if mant & mask == 0:
525//                 shift = diff + 1
526//                 mant <<= shift
527//                 exp -= shift
528//     return mant, exp
529//
530//
531// def round_to_native(mant, exp, finfo):
532//     '''Round float to representation for conversion.'''
533//
534//     mant, exp = normalize(mant, exp, finfo)
535//     mant, exp = round_to_float(mant, exp, finfo)
536//     mant, exp = avoid_overflow(mant, exp, finfo)
537//
538//     return mant, exp
539//
540//
541// def parse_binary(integer, fraction, exponent, finfo):
542//     '''Parses a binary number'''
543//
544//     is_positive = True
545//     if len(integer) > 0 and integer[0] == '-':
546//         integer = integer[1:]
547//         is_positive = False
548//     elif len(integer) > 0 and integer[0] == '+':
549//         integer = integer[1:]
550//     mant = int(integer + fraction, 2)
551//     exp = int(exponent, 2) - len(fraction)
552//     mant, exp = round_to_native(mant, exp, finfo)
553//     return into_float(mant, exp, finfo, is_positive)
554//
555//
556// class f32info:
557//     ftype = np.float32
558//     itype = np.uint32
559//     BITS = 32
560//     SIGN_MASK = 0x80000000
561//     EXPONENT_MASK = 0x7F800000
562//     HIDDEN_BIT_MASK = 0x00800000
563//     MANTISSA_MASK = 0x007FFFFF
564//     INFINITY_BITS = 0x7F800000
565//     MANTISSA_SIZE = 23
566//     EXPONENT_BIAS = 127 + MANTISSA_SIZE
567//     DENORMAL_EXPONENT = 1 - EXPONENT_BIAS
568//     MAX_EXPONENT = 0xFF - EXPONENT_BIAS
569//     DEFAULT_SHIFT = BITS - MANTISSA_SIZE - 1
570//     CARRY_MASK = 0x1000000
571//
572// parse_binary('1', '001111000000110010', '0', f32info) -> 1.2345657
573// parse_binary('1', '001111000000110001', '0', f32info) -> 1.2345619
574// parse_binary('1', '001111000000110011', '0', f32info) -> 1.2345695
575// parse_binary('1', '0011110000001100101', '0', f32info) -> 1.2345676
576// parse_binary('1', '0011110000001100101001', '0', f32info) -> 1.2345679
577// parse_binary('1100', '010110000111111', '0', f32info) -> 12.345673
578// parse_binary('1100', '01011000011111100111', '0', f32info) -> 12.345673
579// ```
580
581// WRITER
582
583// The writer is therefore quite simple:
584//  1). Write the significant digits, using an itoa algorithm for the radix.
585//      We never need to backtrack, because we cannot have rounding error.
586//  2). Calculate the scientific exponent for the value.
587//      This is exponent + mantissa digits.
588//  3). Determine if we should use exponent notation.
589//  4). Write the digits, scaled by the exponent, to the buffer.
590//  5). Write the exponent character and the scientific exponent, if applicable.
591//
592// We can validate a written value (without an exponent)
593// with the following:
594//
595// ```python
596// def to_float(value, radix, exp_base=None, exp_radix=None):
597//     '''Convert a string to float.'''
598//
599//     if exp_base is None:
600//         exp_base = radix
601//     if exp_radix is None:
602//         exp_radix = radix
603//
604//     integer_digits, fraction_digits = value.split('.')
605//     integer = int(integer_digits, radix)
606//
607//     split = fraction_digits.split('^')
608//     if len(split) == 2:
609//         fraction_digits, exponent_digits = split
610//         fraction = int(fraction_digits, radix)
611//         exponent = int(exponent_digits, exp_radix)
612//     else:
613//         fraction = int(fraction_digits, radix)
614//         exponent = 0
615//
616//     mantissa = integer + (fraction * radix**(-len(fraction_digits)))
617//     return mantissa * exp_base**exponent
618// ```

623/// Fast integral log2.
624/// Only to be used on radixes.
625#[inline(always)]
626pub fn fast_log2(x: u32) -> i32 {
627    debug_assert!(matches!(x, 2 | 4 | 8 | 16 | 32));
628    32 - 1 - (x | 1).leading_zeros() as i32
629}

631/// Calculate the number of significant bits in an integer.
632/// This is just `1-ctlz(value)`.
633#[inline(always)]
634pub fn significant_bits<T: UnsignedInteger>(value: T) -> u32 {
635    T::BITS as u32 - value.leading_zeros()
636}

638/// Calculate the fast ceiling, integral division.
639#[inline(always)]
640pub fn fast_ceildiv(value: i32, base: i32) -> i32 {
641    debug_assert!(value >= 0);
642    (value + base - 1) / base
643}

645/// Get the inverse remainder to calculate the modulus.
646///
647/// For negative numbers, the remainder is is essentially:
648///     `⌊-x / y⌋ = z, r = -x - (y * z)`
649///     `⌊-21 / 4⌋ = -5, r = -21 - (4 * -5), or r = -1`
650///
651/// We, however, want the modulus to calculate the left-shift
652/// necessary. The modulus for negative numbers is effectively:
653///     `⌈-x / y⌉ = z, r = -x - (y * z)`
654///     `⌈-21 / 4⌉ = -6, r = -21 - (4 * -6), or r = 3`
655#[inline(always)]
656pub fn inverse_remainder(remainder: i32, base: i32) -> i32 {
657    debug_assert!(remainder >= 0 && remainder < base);
658    match remainder {
659        0 => 0,
660        rem => base - rem,
661    }
662}
663
664/// Calculate the shift to align the highest mantissa bit.
665///
666/// We need to calculate the shift to align the
667/// highest mantissa bit. This is based off aligning
668/// the bits to the exponent, so we can get it on
669/// an exponent boundary. For example, at exp = -55,
670/// we get a shl of 1 from the modulus `(-55 % 4)`.
671/// Since rust doesn't support the modulus operation,
672/// just remainder, we can use the `inverse_remainder`
673/// function to swap between the remainder of the inverse
674/// to the modulus. Likewise, at `(-53 % 4) == 3`,
675/// `-55 % 3 == 2`, and `-53 % 5 == 2`, all numbers
676/// we expect.
677///
678/// For positive numbers, this is quite simple: `(48 % 5) => == 3`,
679/// which is just the modulus operation as well (which for positive
680/// numbers, this is identical to Rust's remainder operator).
681///
682/// NOTE: this can never overflow the significant digits, since
683/// the mantissa must be at **least** 6-bits smaller than the mantissa
684/// size (for `f16`, larger for anything else) and `bits_per_digit <= 5`.
685#[inline(always)]
686pub fn calculate_shl(exp: i32, bits_per_digit: i32) -> i32 {
687    if exp < 0 {
688        let shr = exp.wrapping_neg() % bits_per_digit;
689        inverse_remainder(shr, bits_per_digit)
690    } else {
691        exp % bits_per_digit
692    }
693}
694
695/// We need to scale the scientific exponent for writing.
696///
697/// If we have a negative exp, then when scaling that,
698/// we need to consider that an exp of -1 with 5 bits
699/// per base is still <0, IE, the sci exp we write has
700/// to be: `⌊sci_exp / bits_per_base⌋`, where `ceil` is
701/// wrapping towards greatest magnitude.
702///
703/// If we have a positive exp, we just need the floor of the
704/// `sci_exp / bits_per_base`, because if we had an exp of 1 with
705/// 5 bits, that would behind the decimal point.
706#[inline(always)]
707pub fn scale_sci_exp(sci_exp: i32, bits_per_digit: i32) -> i32 {
708    if sci_exp < 0 {
709        let neg_sci_exp = sci_exp.wrapping_neg();
710        fast_ceildiv(neg_sci_exp, bits_per_digit).wrapping_neg()
711    } else {
712        sci_exp / bits_per_digit
713    }
714}

719/// Round mantissa to the nearest value, returning only the number
720/// of significant digits. Also returns the number of bits of the mantissa.
721#[inline(always)]
722pub fn truncate_and_round<M>(mantissa: M, radix: u32, options: &Options) -> (M, usize)
723where
724    M: UnsignedInteger,
725{
726    let mut mantissa_bits = significant_bits(mantissa) as usize;
727    let bits_per_digit = fast_log2(radix);
728
729    // Get the number of max digits, and then calculate if we need to round.
730    let mut max_digits = usize::MAX;
731    if let Some(digits) = options.max_significant_digits() {
732        max_digits = digits.get();
733    }
734    let max_bits = max_digits.saturating_mul(bits_per_digit as usize);
735    let mut shifted_mantissa = mantissa;
736
737    // Need to truncate the number of significant digits.
738    if max_bits < mantissa_bits {
739        let shr = (mantissa_bits - max_bits) as i32;
740        shifted_mantissa = mantissa >> shr;
741
742        // We need to round-nearest, tie-even, so we need to handle
743        // the truncation **here**. If the representation is above
744        // halfway at all, we need to round up, even if 1 bit.
745        if options.round_mode() == RoundMode::Round {
746            let mask = (M::ONE << shr) - M::ONE;
747            let halfway = M::ONE << (shr - 1);
748            let above_halfway = (mantissa & mask) > halfway;
749            let is_halfway = (mantissa & mask) == halfway;
750            let is_odd = shifted_mantissa & M::ONE == M::ONE;
751
752            // Round-up and calculate if we carry over 1-bit.
753            // The built-in ctlz is very fast, so use that.
754            // Add 1 to the mantissa bits if we carry.
755            let initial_bits = shifted_mantissa.leading_zeros();
756            shifted_mantissa += as_cast((above_halfway || (is_odd & is_halfway)) as u32);
757            let final_bits = shifted_mantissa.leading_zeros();
758            mantissa_bits += (final_bits - initial_bits) as usize;
759        }
760    }
761
762    (shifted_mantissa, mantissa_bits)
763}
